結構

程式碼

<!DOCTYPE html>
<html lang="zh-TW">
<head>
    <!-- 這是標題 -->
    <title>HTML練習</title>
    <!-- mate 元素 -->
    <meta charset="utf-8">
    <!-- 網站描述 -->
    <meta name="description" content="網頁描述">
    <!-- 網站關鍵字 -->
    <meta name="keywords" content="關鍵字A, 關鍵字B">
    <!-- CSS 區塊 -->
    <style>
        h1{color:red;}
    </style>
</head>
	
<body>

<!-- 大標題，頁首區塊 -->
<header>
    <h1>我是大標題</h1>
    <p>我是段落</p>
</header>
    
<!-- 導覽連結區塊 -->
<nav>
    <ul>
        <li><a href="#intro">跳轉區塊一</a></li>
        <li><a href="#scenery">跳轉區塊二</a></li>
        <li><a href="#profile">跳轉區塊三</a></li>
    </ul>
</nav>
        
<!-- 主內容區塊 -->
<main>

<!-- 內容1 -->
<section id="intro">
    <h2>區塊一標題</h2>
    <p>歡迎光臨本站。
    填充文字填充文字填充文字填充文字填充文字。<br>
    <!-- strong 加粗 -->>
    <strong>※未經許可，請勿擅自複製轉載。</strong></p>
</section>
    
<!-- 內容2 -->
<section id="scenery">
    <h2>區塊二標題</h2>
    
    <!-- 內容2-1 -->
    <section>
        <h3>區塊 2-1 標題</h3>
        <img src="img/scenery1.jpg" width="480" height="320" alt="風景1">
        <p>填充文字填充文字填充文字填充文字填充文字。</p>
        <p><a href="scenery/scenery1.html">→更多介紹</a></p>
    </section>

    <!-- 內容2-2 -->
    <section>
        <h3>區塊 2-2 標題</h3>
        <img src="img/scenery2.jpg" width="480" height="320" alt="風景2">
        <p>填充文字填充文字填充文字填充文字填充文字。</p>
        <p><a href="scenery/scenery2.html">→更多介紹</a></p>
    </section>
</section>

<!-- 內容3 -->
<section id="profile">
    <h2>區塊三標題</h2>
    <img src="img/avatar.png" width="250" height="250" alt="大頭照">
    <dl>
        <dt>列表標題1 ：</dt><dd>列表資料1</dd>
        <dt>列表標題2 ：</dt><dd>列表資料2</dd>
        <dt>列表標題3 ：</dt><dd><a href="mailto:xxxx1234@gmail.com">xxxx1234@gmail.com</a></dd>
        <dt>列表標題4 ：</dt><dd><a href="" target="_blank">http://xxx</a></dd>
    </dl>
</section>
</main>    

<!-- 頁尾區塊 -->
<footer>
    <!-- small 標籤常使用版權宣告 -->
    <p><small>LICENSED UNDER CC BY-NC-SA 4.0</small></p>
</footer>

</body>
</html>
